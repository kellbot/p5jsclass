The draw() loop


							let yPos = 0;
							function setup() {
							  // setup() runs once
							  frameRate(30);
							}
							function draw() {
							  // draw() loops forever
							  background(204);
							  yPos = yPos - 1;
							  if (yPos < 0) {
							    yPos = height;
							  }
							  line(0, yPos, width, yPos);
							}

							function setup() {
							  createCanvas(400, 400);
							  noLoop();
							}

							function draw() {
							  let bg_r = random(0,255);
							  let bg_g = random(0,255);
							  let bg_b = random(0,255);
							  background(bg_r, bg_g, bg_b ); 
							}
